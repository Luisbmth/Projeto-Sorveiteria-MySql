create database dexprag;
use dexprag;

create table Cliente(
id int(11) auto_increment primary key not null,
nome varchar (100) not null,
cpf_cnpj bigint(14) not null,
endereço varchar(300) not null,
cep int (8) not null
);

create table Serviços(
id int(11) auto_increment primary key not null,
data date not null,
hora time not null,
coleta_lixo boolean not null,
area_tratada boolean,
fk_id int(11) not null,
foreign key (fk_id) references Cliente(id)
);

create table Praga(
id int(11) auto_increment primary key not null,
baratas boolean,
rato boolean, 
cupim boolean,
aranha boolean, 
forminga boolean,
pombo boolean,
outro varchar(30),
fk_id int(14) not null,
foreign key (fk_id) references Serviços(id)
);


create table Produtos(
id int(14) auto_increment primary key not null,
descriçao varchar (60) not null,
dosagem float(3,2) not null,
veiculo varchar(25) not null,
quantidade int(3) not null,
alvo varchar(30) not null,
fk_id int(11) not null,
foreign key (fk_id) references Praga(id)
);

create table usuario(
id int(11) auto_increment primary key not null,
nome varchar(100) not null,
email varchar(100) unique not null ,
senha varchar(255) not null
);
